                                                # About Array

                                                When memory is allocated for an array, its individual elements are adjacent, so we can access it by index.

                                                let nums = [10, 11, 12, 13, 14, 15];

Array:   10  11  12  13  14  15
index:    0   1   2   3   4   5

assert_eq(nums[0], 10);
assert_eq(nums[5], 15);
assert_eq(nums[6], xx); // ERROR，index out of bounds
